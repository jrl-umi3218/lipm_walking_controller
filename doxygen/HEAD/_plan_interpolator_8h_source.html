166  void gait(const std::string & dir)
167  {
168  if(dir == "Shuffle")
169  {
170  extraStepWidth_ = DEFAULT_EXTRA_STEP_WIDTH;
171  gait_ = Gait::Shuffle;
172  }
173  else if(dir == "Turn")
174  {
175  extraStepWidth_ = IN_PLACE_EXTRA_STEP_WIDTH;
176  gait_ = Gait::Turn;
177  targetPose_.x = 0.;

179  }
180  else // (dir == "Walk")
181  {
182  extraStepWidth_ = DEFAULT_EXTRA_STEP_WIDTH;
183  gait_ = Gait::Walk;
184  }
185  run();
186  }
