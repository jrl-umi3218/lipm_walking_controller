248  void wrenchFaceMatrix(const Sole & sole)
249  {
250  double X = sole.halfLength;
251  double Y = sole.halfWidth;
252  double mu = sole.friction;
253  // clang-format off
254  wrenchFaceMatrix_ <<
255  // mx, my, mz, fx, fy, fz,
256  0, 0, 0, -1, 0, -mu,
257  0, 0, 0, +1, 0, -mu,
258  0, 0, 0, 0, -1, -mu,
259  0, 0, 0, 0, +1, -mu,
260  -1, 0, 0, 0, 0, -Y,
261  +1, 0, 0, 0, 0, -Y,
262  0, -1, 0, 0, 0, -X,
263  0, +1, 0, 0, 0, -X,
264  +mu, +mu, -1, -Y, -X, -(X + Y) * mu,
265  +mu, -mu, -1, -Y, +X, -(X + Y) * mu,
266  -mu, +mu, -1, +Y, -X, -(X + Y) * mu,
267  -mu, -mu, -1, +Y, +X, -(X + Y) * mu,
268  +mu, +mu, +1, +Y, +X, -(X + Y) * mu,
269  +mu, -mu, +1, +Y, -X, -(X + Y) * mu,
270  -mu, +mu, +1, -Y, +X, -(X + Y) * mu,
271  -mu, -mu, +1, -Y, -X, -(X + Y) * mu;
272  // clang-format on
273  }
