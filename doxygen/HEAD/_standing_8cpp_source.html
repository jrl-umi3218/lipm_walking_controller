41 void states::Standing::start()
42 {
43  auto & ctl = controller();
44  auto & supportContact = ctl.supportContact();
45  auto & targetContact = ctl.targetContact();
46 
47  freeFootGain_ = 30.;
48  isMakingFootContact_ = false;
49  planChanged_ = false;
50  lastInterpolatorIter_ = ctl.planInterpolator.nbIter;
51  leftFootRatio_ = ctl.leftFootRatio();
52  releaseHeight_ = 0.05; // [m]
53  startWalking_ = ctl.config()("autoplay", false);
54  if(supportContact.surfaceName == "RightFootCenter")
55  {
56  leftFootContact_ = targetContact;
57  rightFootContact_ = supportContact;
58  }
59  else if(supportContact.surfaceName == "LeftFootCenter")
60  {
61  leftFootContact_ = supportContact;
62  rightFootContact_ = targetContact;
63  }
64  else
65  {
66  mc_rtc::log::error_and_throw<std::invalid_argument>("Unknown surface name: {}", supportContact.surfaceName);
67  }
68 
69  if(ctl.isLastDSP())
70  {
71  ctl.loadFootstepPlan(ctl.plan.name);
72  }
73 
74  stabilizer().contactState(ContactState::DoubleSupport);
75  stabilizer().setContact(stabilizer().leftFootTask, leftFootContact_);
76  stabilizer().setContact(stabilizer().rightFootTask, rightFootContact_);
77  stabilizer().addTasks(ctl.solver());
78  ctl.solver().addTask(ctl.pelvisTask);
79  ctl.solver().addTask(ctl.torsoTask);
80 
81  updatePlan(ctl.plan.name);
82  updateTarget(leftFootRatio_);
83 
84  logger().addLogEntry("support_xmax",
85  [&ctl]() { return std::max(ctl.supportContact().xmax(), ctl.targetContact().xmax()); });
86  logger().addLogEntry("support_xmin",
87  [&ctl]() { return std::min(ctl.supportContact().xmin(), ctl.targetContact().xmin()); });
88  logger().addLogEntry("support_ymax",
89  [&ctl]() { return std::max(ctl.supportContact().ymax(), ctl.targetContact().ymax()); });
90  logger().addLogEntry("support_ymin",
91  [&ctl]() { return std::min(ctl.supportContact().ymin(), ctl.targetContact().ymin()); });
92  logger().addLogEntry("support_zmax",
93  [&ctl]() { return std::max(ctl.supportContact().zmax(), ctl.targetContact().zmax()); });
94  logger().addLogEntry("support_zmin",
95  [&ctl]() { return std::min(ctl.supportContact().zmin(), ctl.targetContact().zmin()); });
96  logger().addLogEntry("walking_phase", []() { return 3.; });
97  ctl.stopLogSegment();
98 
99  if(startWalking_) // autoplay
100  {
101  auto plans = ctl.config()("autoplay_plans", std::vector<std::string>{});
102  if(plans.size() == 0)
103  {
104  startWalking_ = false;
105  ctl.config().add("autoplay", false);
106  }
107  else
108  {
109  std::string plan = plans[0];
110  plans.erase(plans.begin());
111  ctl.config().add("autoplay_plans", plans);
112  lastInterpolatorIter_++;
113  updatePlan(plan);
114  }
115  }
116 
117  if(gui())
118  {
119  using namespace mc_rtc::gui;
120  gui()->removeElement({"Walking", "Main"}, "Pause walking");
121  gui()->addElement({"Walking", "Main"}, ComboInput("Footstep plan", ctl.planInterpolator.availablePlans(),
122  [&ctl]() { return ctl.plan.name; },
123  [this](const std::string & name) { updatePlan(name); }));
124  gui()->addElement({"Walking", "Main"}, Button((supportContact.id == 0) ? "Start walking" : "Resume walking",
125  [this]() { startWalking(); }));
126  gui()->addElement({"Standing"},
127  NumberInput("CoM target [0-1]", [this]() { return std::round(leftFootRatio_ * 10.) / 10.; },
128  [this](double leftFootRatio) { updateTarget(leftFootRatio); }),
129  NumberInput("Free foot gain", [this]() { return std::round(freeFootGain_); },
130  [this](double gain) { freeFootGain_ = clamp(gain, 5., 100.); }),
131  NumberInput("Release height [m]", [this]() { return std::round(releaseHeight_ * 100.) / 100.; },
132  [this](double height) { releaseHeight_ = clamp(height, 0., 0.25); }),
133  Label("Left foot force [N]",
134  [&ctl]() { return ctl.realRobot().forceSensor("LeftFootForceSensor").force().z(); }),
135  Label("Right foot force [N]",
136  [&ctl]() { return ctl.realRobot().forceSensor("RightFootForceSensor").force().z(); }),
137  Button("Go to left foot", [this]() { updateTarget(1.); }),
138  Button("Go to middle", [this]() { updateTarget(0.5); }),
139  Button("Go to right foot", [this]() { updateTarget(0.); }),
140  Button("Make left foot contact", [this]() { makeLeftFootContact(); }),
141  Button("Make right foot contact", [this]() { makeRightFootContact(); }),
142  Button("Release left foot", [this]() { releaseLeftFootContact(); }),
143  Button("Release right foot", [this]() { releaseRightFootContact(); }));
144  }
145 
146  runState(); // don't wait till next cycle to update reference and tasks
147 }

174 void states::Standing::runState()
175 {
176  checkPlanUpdates();
177 
178  if(isMakingFootContact_)
179  {
180  auto & leftFootTask = stabilizer().leftFootTask;
181  auto & rightFootTask = stabilizer().rightFootTask;
182  bool isLeftFootSeekingContact = (leftFootTask->admittance().couple().x() < 1e-10);
183  bool isRightFootSeekingContact = (rightFootTask->admittance().couple().x() < 1e-10);
184  bool leftFootTouchdown = stabilizer().detectTouchdown(leftFootTask, leftFootContact_);
185  bool rightFootTouchdown = stabilizer().detectTouchdown(rightFootTask, rightFootContact_);
186  if(isLeftFootSeekingContact && leftFootTouchdown)
187  {
188  stabilizer().setContact(leftFootTask, leftFootContact_);
189  }
190  if(isRightFootSeekingContact && rightFootTouchdown)
191  {
192  stabilizer().setContact(rightFootTask, rightFootContact_);
193  }
194  isLeftFootSeekingContact = (leftFootTask->admittance().couple().x() < 1e-10);
195  isRightFootSeekingContact = (rightFootTask->admittance().couple().x() < 1e-10);
196  if(!isLeftFootSeekingContact && !isRightFootSeekingContact)
197  {
198  isMakingFootContact_ = false;
199  stabilizer().contactState(ContactState::DoubleSupport);
200  }
201  }
202 
203  auto & ctl = controller();
204  auto & pendulum = ctl.pendulum();
205 
206  Eigen::Vector3d comTarget = copTarget_ + Eigen::Vector3d{0., 0., ctl.plan.comHeight()};
207  const Eigen::Vector3d & com_i = pendulum.com();
208  const Eigen::Vector3d & comd_i = pendulum.comd();
209  const Eigen::Vector3d & cop_f = copTarget_;
210 
211  double K = COM_STIFFNESS;
212  double D = 2 * std::sqrt(K);
213  Eigen::Vector3d comdd = K * (comTarget - com_i) - D * comd_i;
214  Eigen::Vector3d n = ctl.supportContact().normal();
215  double lambda = n.dot(comdd - world::gravity) / n.dot(com_i - cop_f);
216  Eigen::Vector3d zmp = com_i + (world::gravity - comdd) / lambda;
217 
218  pendulum.integrateIPM(zmp, lambda, ctl.timeStep);
219  ctl.leftFootRatio(leftFootRatio_);
220  ctl.stabilizer().run();
221 }

244 void states::Standing::updateTarget(double leftFootRatio)
245 {
246  auto & sole = controller().sole();
247  if(controller().stabilizer().contactState() != ContactState::DoubleSupport)
248  {
249  mc_rtc::log::error("Cannot update CoM target while in single support");
250  return;
251  }
252  leftFootRatio = clamp(leftFootRatio, 0., 1., "Standing target");
253  sva::PTransformd X_0_lfr =
254  sva::interpolate(rightFootContact_.anklePose(sole), leftFootContact_.anklePose(sole), leftFootRatio);
255  copTarget_ = X_0_lfr.translation();
256  leftFootRatio_ = leftFootRatio;
257 }

259 void states::Standing::makeFootContact(std::shared_ptr<mc_tasks::force::CoPTask> footTask, const Contact & contact)
260 {
261  auto & stabilizer = controller().stabilizer();
262  if(footTask->admittance().couple().x() > 1e-10 || stabilizer.detectTouchdown(footTask, contact))
263  {
264  mc_rtc::log::warning("Foot is already in contact");
265  return;
266  }
267  stabilizer.setSwingFoot(footTask);
268  stabilizer.seekTouchdown(footTask);
269  footTask->stiffness(freeFootGain_); // sets damping as well
270  footTask->targetPose(contact.pose);
271  isMakingFootContact_ = true;
272 }

284 bool states::Standing::releaseFootContact(std::shared_ptr<mc_tasks::force::CoPTask> footTask)
285 {
286  constexpr double MAX_FOOT_RELEASE_FORCE = 50.; // [N]
287  auto & stabilizer = controller().stabilizer();
288  if(footTask->admittance().couple().x() < 1e-10)
289  {
290  mc_rtc::log::warning("Foot contact is already released");
291  return false;
292  }
293  else if(footTask->measuredWrench().force().z() > MAX_FOOT_RELEASE_FORCE)
294  {
295  mc_rtc::log::error("Contact force is too high to release foot");
296  return false;
297  }
298  sva::PTransformd X_0_f = footTask->surfacePose();
299  sva::PTransformd X_f_t = Eigen::Vector3d{0., 0., releaseHeight_};
300  stabilizer.setSwingFoot(footTask);
301  footTask->stiffness(freeFootGain_); // sets damping as well
302  footTask->targetPose(X_f_t * X_0_f);
303  return true;
304 }

322 bool states::Standing::checkTransitions()
323 {
324  auto & ctl = controller();
325 
326  if(isMakingFootContact_)
327  {
328  return false;
329  }
330  if(!startWalking_)
331  {
332  return false;
333  }
334 
335  ctl.mpc().contacts(ctl.supportContact(), ctl.targetContact(), ctl.nextContact());
336  ctl.mpc().phaseDurations(0., ctl.plan.initDSPDuration(), ctl.singleSupportDuration());
337  if(ctl.updatePreview())
338  {
339  ctl.nextDoubleSupportDuration(ctl.plan.initDSPDuration());
340  ctl.startLogSegment(ctl.plan.name);
341  output("DoubleSupport");
342  return true;
343  }
344  return false;
345 }

360 void states::Standing::updatePlan(const std::string & name)
361 {
362  auto & ctl = controller();
363  if(name.find("custom") != std::string::npos)
364  {
365  if(!ctl.planInterpolator.isShown)
366  {
367  ctl.planInterpolator.addGUIElements();
368  ctl.planInterpolator.isShown = true;
369  }
370  if(name.find("backward") != std::string::npos)
371  {
372  ctl.planInterpolator.restoreBackwardTarget();
373  }
374  else if(name.find("forward") != std::string::npos)
375  {
376  ctl.planInterpolator.restoreForwardTarget();
377  }
378  else if(name.find("lateral") != std::string::npos)
379  {
380  ctl.planInterpolator.restoreLateralTarget();
381  }
382  ctl.loadFootstepPlan(ctl.planInterpolator.customPlanName());
383  }
384  else // new plan is not custom
385  {
386  if(ctl.planInterpolator.isShown)
387  {
388  ctl.planInterpolator.removeGUIElements();
389  ctl.planInterpolator.isShown = false;
390  }
391  ctl.loadFootstepPlan(name);
392  }
393  planChanged_ = true;
394 }
